<script>
  (() => {
    const players = Array.from(document.querySelectorAll(".bilibili-player"));
    if (!players.length) return;

    const getIframe = (player) => player.querySelector("iframe");
    const getButton = (player) => player.querySelector(".video-activate-btn");

    const unloadPlayer = (player) => {
      const iframe = getIframe(player);
      if (!iframe || !iframe.getAttribute("src")) return;
      iframe.removeAttribute("src");
      player.classList.remove("is-active");
    };

    const loadPlayer = (player) => {
      const iframe = getIframe(player);
      if (!iframe) return;
      const embed = player.dataset.embed;
      if (!embed) return;
      if (iframe.getAttribute("src") === embed) {
        player.classList.add("is-active");
        return;
      }
      iframe.setAttribute("src", embed);
      player.classList.add("is-active");
    };

    const activatePlayer = (player) => {
      players.forEach((item) => {
        if (item === player) {
          loadPlayer(item);
        } else {
          unloadPlayer(item);
        }
      });
    };

    const activateCenteredPlayer = () => {
      let closest = null;
      let distance = Number.POSITIVE_INFINITY;
      const viewportCenter = window.innerHeight / 2;

      players.forEach((player) => {
        const card = player.closest(".video-item");
        if (!card) return;
        const rect = card.getBoundingClientRect();
        const visible = rect.bottom > 80 && rect.top < window.innerHeight - 80;
        if (!visible) return;
        const cardCenter = rect.top + rect.height / 2;
        const currentDistance = Math.abs(cardCenter - viewportCenter);
        if (currentDistance < distance) {
          distance = currentDistance;
          closest = player;
        }
      });

      if (closest) activatePlayer(closest);
    };

    let rafId = null;
    const queueCenterActivation = () => {
      if (rafId) cancelAnimationFrame(rafId);
      rafId = requestAnimationFrame(activateCenteredPlayer);
    };

    players.forEach((player) => {
      const button = getButton(player);
      const card = player.closest(".video-item");

      if (button) {
        button.addEventListener("click", () => activatePlayer(player));
      }

      if (card) {
        card.addEventListener("mouseenter", () => activatePlayer(player));
        card.addEventListener("mouseleave", queueCenterActivation);
      }
    });

    window.addEventListener("scroll", queueCenterActivation, { passive: true });
    window.addEventListener("resize", queueCenterActivation);
    queueCenterActivation();
  })();
</script>
